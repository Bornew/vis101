<script type = "text/javascript" src = "d3/d3.v3.j3"></script> <!-- 调用d3.js库 --> 
 
<script> 
 
 
    //写D3.js代码 
    /** 
     *需要实现什么？ 
     * 1.画一块画布，这个画布：长宽，边距， 
     * 
     */ 
 
    var size = 230, //size是长度和宽度 
        padding = 20; //设定画布大小 230/6  = 38。3 
 
    var x = d3.scale.linear() 
        .range([padding / 2, size - padding / 2]); //设定值域；x的范围 
 
    var y = d3.scale.linear() 
        .range([size - padding / 2, padding / 2]); //y的范围 
 
    var xAxis = d3.svg.axis() //先定义这个坐标轴 
        .scale(x) 
        .orient("bottom") 
        .ticks(6); 
 
    var yAxis = d3.svg.axis() 
        .scale(y) //设定比例尺 
        .orient("left") //设定坐标轴方向 
        .ticks(6); //设定分隔数 
 
    var color = d3.scale.category10(); //能够输出10种类别颜色的预设序数比例尺 
 
    d3.csv("flowers.csv", function(error, data) { //接下来是对数据做处理和呈现工作 
        if (error) throw error; 
 
        var domainByTrait = {}, 
            traits = d3.keys(data[0]).filter(function(d) { return d !== "species"; }), //就是第一条数据，包含了所有数据类型，但是species不算。得到的是横纵轴需要的那几个维度：花萼高度，宽度；花瓣高度，宽度。 
            n = traits.length; //总共四个维度。n = 4; 
 
        traits.forEach(function(trait) { 
            domainByTrait[trait] = d3.extent(data, function(d) { return d[trait]; }); 
        }); 
 
        xAxis.tickSize(size * n); //axis.tickSize([inner,outer])设定或获取坐标轴内外刻度的长度。默认是6 
        yAxis.tickSize(-size * n); 
 
        var svg = d3.select("body").append("svg") //选择body元素，再往里面创建一个svg元素，然后把该设定的宽度高度以及该放上去的坐标轴属性赋给它 
            .attr("width", size * n + padding) 
            .attr("height", size * n + padding) 
            .append("g") 
            .attr("transform", "translate(" + padding + "," + padding / 2 + ")"); //将原点放到指定位置 
 
        svg.selectAll(".x.axis" //可以理解成要添加很多个x.axis，先预定这些位置，等到有了数据了（也就是下面这条data（traits）相当于是在取其中数据），然后插入svg容器中 
            .data(traits) //？？？绑定数据，即traits对应的数据 
            .enter().append("g") //enter就表示有数据来了，将每个数据添加到单独的"g"里面去 
            .attr("class", "x axis")  //往"g"里面赋值，添加一个类，叫做"x axis" 
            .attr("transform", function(d, i) { return "translate(" + (n - i - 1) * size + ",0)"; }) //给transform赋值函数 该函数的作用是将光标移动到n-1-i ,0 
            .each(function(d) { x.domain(domainByTrait[d]); d3.select(this).call(xAxis); }); //将选择的元素（this分别指向各x.axis）再将它每一个传给xAxis函数 
 
        svg.selectAll(".y.axis") 
            .data(traits) 
            .enter().append("g") 
            .attr("class", "y axis") 
            .attr("transform", function(d, i) { return "translate(0," + i * size + ")"; }) 
            .each(function(d) { y.domain(domainByTrait[d]); d3.select(this).call(yAxis); }); 
 
        var cell = svg.selectAll(".cell") 
            .data(cross(traits, traits)) //函数见最后 ，就是搞了一个数组出来，之后会按照这个数组来建立需要的格子 
            .enter().append("g") 
            .attr("class", "cell") 
            .attr("transform", function(d) { return "translate(" + (n - d.i - 1) * size + "," + d.j * size + ")"; }) //移动光标到 
            .each(plot); 
 
        // Titles for the diagonal. 
        cell.filter(function(d) { return d.i === d.j; }).append("text") //只有那些横纵坐标相等的会有文字显示 
            .attr("x", padding) 
            .attr("y", padding) 
            .attr("dy", ".71em") 
            .text(function(d) { return d.x; }); 
 
        function plot(p) { 
            var cell = d3.select(this); //画点 
 
            x.domain(domainByTrait[p.x]); //作用域？？ 
            y.domain(domainByTrait[p.y]); 
 
            cell.append("rect") 
                .attr("class", "frame") 
                .attr("x", padding / 2) 
                .attr("y", padding / 2) 
                .attr("width", size - padding) 
                .attr("height", size - padding); 
 
            cell.selectAll("circle")  //创建这么多circle，读取data，然后给他们圆心和半径，加到svg上 
                .data(data) 
                .enter().append("circle") 
                .attr("cx", function(d) { return x(d[p.x]); }) 
                .attr("cy", function(d) { return y(d[p.y]); }) 
                .attr("r", 4) 
                .style("fill", function(d) { return color(d.species); }); 
        } 
    }); 
 
    function cross(a, b) { 
        var c = [], n = a.length, m = b.length, i, j; 
        for (i = -1; ++i < n;) for (j = -1; ++j < m;) c.push({x: a[i], i: i, y: b[j], j: j}); 
        return c;//传入这个数组的是：【x：花萼高度，i：0，y：花萼高度，j：0】 以此类推 总共4*4 共16个 
    } 
 
</script>
